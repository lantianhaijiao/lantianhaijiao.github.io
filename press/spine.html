function traverseDir(dir: string) {
  const files = fs.readdirSync(dir);
  files.forEach((file: any) => {
    const filePath = path.join(dir, file);

      if (ext === ".json") {
        jsonFiles.push(filePath);
      } else if (ext === ".atlas") {
        atlasFiles.push(filePath);
      }
    }
  });
